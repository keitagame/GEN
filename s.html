CPUコアの基本実装

class M68K {
  constructor() {
    this.d = new Int32Array(8);   // D0-D7
    this.a = new Uint32Array(8);  // A0-A7
    this.pc = 0;
    this.sr = 0x2700;             // スーパーバイザーモード、割り込みマスク=7
    this.usp = 0;
    this.cycles = 0;
  }

  reset(bus) {
    // リセットベクタからSP/PCを読み込む
    this.a[7] = bus.read32(0x000000);
    this.pc   = bus.read32(0x000004);
    this.sr   = 0x2700;
  }

  step(bus) {
    const opcode = bus.read16(this.pc);
    this.pc += 2;
    return this.execute(opcode, bus);
  }
}

function setVDPAddress(vdp, addr, type) {
  const w1 = ((type & 3) << 14) | (addr & 0x3FFF);
  const w2 = ((type & 0x3C) << 2) | ((addr >> 14) & 3);
  vdp.writeControl(w1);
  vdp.writeControl(w2);
}

CRAMカラーフォーマット

// CRAM エントリ = 16bit
// [11:9] 青 (B) 3bit → 0,36,73,109,146,182,219,255
// [7:5]  緑 (G) 3bit
// [3:1]  赤 (R) 3bit

function cramToRGBA(cram_entry) {
  const r = (cram_entry >> 1) & 7;
  const g = (cram_entry >> 5) & 7;
  const b = (cram_entry >> 9) & 7;
  // 3bit → 8bit 変換 (0,36,73,109,146,182,219,255)
  const lut = [0, 36, 73, 109, 146, 182, 219, 255];
  return [lut[r], lut[g], lut[b], 255];
}

Web Audio API でのサウンド実装

class AudioSystem {
  constructor() {
    this.ctx = new AudioContext({ sampleRate: 44100 });
    this.bufferSize = 4096;
    this.ym2612 = new YM2612();
    this.psg = new SN76489();
  }

  createNode() {
    this.processor = this.ctx.createScriptProcessor(
      this.bufferSize, 0, 2
    );
    this.processor.onaudioprocess = (e) => {
      const left  = e.outputBuffer.getChannelData(0);
      const right = e.outputBuffer.getChannelData(1);
      for (let i = 0; i < this.bufferSize; i++) {
        const [l, r] = this.ym2612.sample();
        const p = this.psg.sample();
        left[i]  = (l + p) / 32768;
        right[i] = (r + p) / 32768;
      }
    };
    this.processor.connect(this.ctx.destination);
  }
}

メインループ設計

class MegaDrive {
  frameStep() {
    const LINES = 262;
    const M68K_PER_LINE = 488;
    const Z80_PER_LINE  = 228;

    for (let line = 0; line < LINES; line++) {
      // H-INT カウンタ処理
      if (line < 224) {
        if (this.hintCounter-- <= 0) {
          this.hintCounter = this.vdp.reg[10];
          this.m68k.interrupt(4); // H-INT
        }
      }

      // M68K 実行
      this.m68k.runCycles(M68K_PER_LINE);

      // Z80 実行
      if (!this.z80Busreq)
        this.z80.runCycles(Z80_PER_LINE);

      // スキャンライン描画
      if (line < 224)
        this.vdp.renderLine(line);

      // V-Blank 開始
      if (line === 224) {
        this.vdp.setVBlank(true);
        this.m68k.interrupt(6); // V-INT
        this.vdp.presentFrame(this.canvas);
      }
    }
    this.vdp.setVBlank(false);
  }

  run() {
    const targetFPS = 59.923;
    const loop = (timestamp) => {
      this.frameStep();
      requestAnimationFrame(loop);
    };
    requestAnimationFrame(loop);
  }
}

      this.canvas.width, this.canvas.height
    );
  }
}

ROM ローダー

async function loadROM(file) {
  const buffer = await file.arrayBuffer();
  const rom = new Uint8Array(buffer);

  // SMD形式 (インターリーブ) を検出・変換
  if (isSMD(rom)) return decodeSMD(rom);

  // ROMヘッダー検証 (0x100-0x1FF)
  const systemType = readAscii(rom, 0x100, 16); // "SEGA MEGA DRIVE"
  const checksum = (rom[0x18E] << 8) | rom[0x18F];

  // チェックサム計算 (0x200から)
  let calc = 0;
  for (let i = 0x200; i < rom.length; i += 2)
    calc = (calc + ((rom[i] << 8) | rom[i+1])) & 0xFFFF;

  return rom;
}

// ROM ヘッダー構造
// 0x100-0x10F: システム名 ("SEGA MEGA DRIVE ")
// 0x110-0x11F: 著作権/年
// 0x120-0x14F: 国内タイトル
// 0x150-0x17F: 海外タイトル

既存OSS CPUコアの活用

✓ 推奨: CPUコアはゼロから書くより、実績あるOSSライブラリを使用することを強く推奨する。
• M68K: musashi (C→Emscripten/WASM) または m68k.js
• Z80: Z80.js (Floooh/chips の JS移植)
• YM2612: gens-ym2612 の JS移植
これにより開発工数を大幅に削減でき、VDP/バス実装に集中できる。
